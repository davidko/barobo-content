Lesson 6

The Linkbot will measure your reaction time. The LED will change color from red to green and the "A" button is depressed as soon as possible.

Objective

The Linkbot will measure your reaction time. The LED will change color from red to green and the "B" button is depressed as soon as possible. 

import time
import barobo
import random
import sys
dongle = barobo.Dongle()
dongle.connect()
robotID = input('Enter robot ID: ')
robot = dongle.getLinkbot(robotID) #enter the linkbot ID
time_start=0
game_start=False
def callback(mask, buttons, data):
    if buttons & 0x02:
        global game_start
        global time_start
        if game_start==False:
            print("You pressed the button to soon")
            sys.exit()
        else:
            print("your reaction time is :", time.time()-time_start)
        global robot
        robot.setBuzzerFrequency(0)
        sys.exit()
robot.enableButtonCallback(callback)
input("Press enter to begin")
robot.setLEDColor(255, 0, 0)
time.sleep(random.random()*7+3)
time_start=time.time()
game_start=True
robot.setBuzzerFrequency(440)
robot.setLEDColor(0, 255, 0)
while True:
    time.sleep(1)
